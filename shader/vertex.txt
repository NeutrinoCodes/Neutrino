#version 410 core

layout (location = 0) in  vec4 in_coords;                                       // First attribute variable (0).
layout (location = 1) in  vec4 in_color;                                        // Second attribute variable (1).
smooth out vec4 the_color;                                                      // (R, G, B, A) color,

void main(void)
{
    gl_Position = in_coords;                                                    // Assigning "in_coords" as position for OpenGL...
    the_color = in_color;                                                       // Forwarding the "in_color" to the fragment shader...
}
