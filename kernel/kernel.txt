__kernel void pippo(__global float4*  points, int num_points, float tick)
{
  float       x;
  float       x_min = -1.00f;
  float       x_max = 1.00f;

  int i = get_global_id(0);

  x = ((x_max - x_min)/(float)num_points)*(float)i + x_min;
  points[i] = (float4)(x, sin(10*x + 2*tick), 0.00f, 1.00f);
}
