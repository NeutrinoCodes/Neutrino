__kernel void thekernel(__global  float4*       points,
                        __global  float4*       colors,
                        unsigned  int           points_size)
{
  int i = get_global_id(0);
  float4 P = points[i];

  P.z = 0.1*sin(10.0f*P.x) + 0.1*cos(10.0f*P.y);
  points[i] = P;
  barrier(CLK_GLOBAL_MEM_FENCE);
}
